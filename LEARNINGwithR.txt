## EQUALITY

$$
\begin{equation}
\text{ Let } = \text{ represent parity in social participation } \text { ? }
\end{equation}
$$


```{r, echo=FALSE}
rm(list = ls())

I <- 1000000

M <- data.frame(W = c(1),
                X = c((1:I)/I),
                Y = c(1/2),
                Z = c(0))
L <- length(M$X)
XL <- M$X * L
z <- (-1)^(M$X*I)

M.XL <- rbind(head(XL),tail(XL))
M.XL
plot(x = M$X, y = M$Y*z)
```
